g = 10 m/s ²

h = 70 Cm - 25 Cm = 45 Cm = 0,45 Meter

Ditanyakan = tekanan hidrostatis

jawab =

[tex]\displaystyle P_h = \rho \times g\times h \\P_h = 1000\times 10 \times 0,45 \\P_h = 1000 \times 4,5\\P_h = 4500 Pascal[/tex]
